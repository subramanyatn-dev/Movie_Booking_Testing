===============================================
1. CLEAN PROJECT (Always run first)
===============================================
mvn clean

===============================================
2. RUN ALL TESTS (Single Command)
===============================================
mvn test

===============================================
3. RUN INDIVIDUAL TEST CLASSES
===============================================
# Authentication Tests (5 tests)
mvn test -Dtest=AuthenticationTests

# Booking Tests (7 tests)
mvn test -Dtest=BookingTests

# View Booking Tests (7 tests)
mvn test -Dtest=ViewBookingTests

# Delete Booking Tests (6 tests)
mvn test -Dtest=DeleteBookingTests

===============================================
4. RUN MULTIPLE SPECIFIC TESTS
===============================================
# Run Authentication + View Tests
mvn test -Dtest=AuthenticationTests,ViewBookingTests

# Run only passing tests (skip failing ones)
mvn test -Dtest=AuthenticationTests,ViewBookingTests,DeleteBookingTests

# Run by test groups
mvn test -Dgroups=authentication
mvn test -Dgroups=booking
mvn test -Dgroups=viewBooking
mvn test -Dgroups=deleteBooking

===============================================
5. REPORT GENERATION
===============================================

BASIC TESTNG REPORT:
# Report automatically generated after tests
open target/surefire-reports/index.html

ADVANCED ALLURE REPORT:
# Generate report files
mvn allure:report

open target/site/allure-maven-plugin/index.html

# Start server and view report in browser
mvn allure:serve
